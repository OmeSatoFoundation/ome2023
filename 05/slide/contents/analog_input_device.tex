\begin{frame}[fragile]
    \frametitle{アナログ入力を使った\\プログラム}
    \begin{lstlisting}[title=\textasciitilde/05/anain.hsp]
    #include "hsp3dish.as"
    #include "rpz-gpio.as"

    spiopen 0

    *main
	    data = adcget(0,0)
	    res = "結果 : "+data+"\n"

	    redraw 0
	    pos 20,20
	    font "",30
	    mes res
	    redraw 1

	    wait 10
	    goto *main

    spiclose 0
    \end{lstlisting}
    \textpageref{16}
\end{frame}
